import {casFetchRows} from '@sassoftware/restaflib/dist/restaflib.js';
import prepFormData from "./prepFormData";
/**
 * @description Simplify scrolling using next|prev|top
 * @sync
 * @function scrollTable
 * @param {string} scroll direction(next|prev|top)
 * @param {appEnv} appEnv 
 * @returns {promise}  result ready for display or null if it did not scroll
 * @example
 *  let r = await scrollTable('next', appEnv);
 *  if ( r === null) {
 *     handle redisplaying current data
 *  } else {
 *     handle new data
 * }
 */
async function scrollTable (direction,appEnv) {
    const {store, session} = appEnv;
    const {pagination, initialFetch, table} = appEnv.dataControl;
    let control;
    if (direction === 'top') {
        control = {...initialFetch};
        control.table = table;
    } else {
        control = pagination[direction];
    }
 
    if (control.next === -1) {
        return null;
    }
    let r = await casFetchRows(store, session, control);
    let t = await prepFormData(r.data, appEnv);
    appEnv.runTime = {
        modified  : [],
        pagination: {...r.pagination},
        data      : t.rowsObject,
        columns   : t.columns
    };
   
    t.pagination = {...r.pagination};
    return  t;
        
}
export default scrollTable;
